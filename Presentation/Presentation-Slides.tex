\section{Introduction}
\begin{frame}{Motivations \& Research Goals}
    \begin{block}{Primary Goal}
        Investigate properties of \textbf{Probabilistic Principal Component Analysis (PPCA)} and \textbf{Gaussian Process Latent Variable Models (GP-LVM)}
    \end{block}

    \bigskip

    PPCA \citep{PPCA-1999} and GP-LVM \citep{GPLVM-2005} are both extensions of principal component analysis which search for latent variable representations of high-dimensional data.
\end{frame}

\begin{frame}{Research Question}
    \begin{block}{Research Question}
        Which latent variable methods are best suited for reducing the dimensionality of image data?
    \end{block}
  \bigskip

  \textbf{Performance metrics:}
  \begin{itemize}[label=\ding{118}]
    \item \alert{Cluster–Label Agreement} between clusters in latent variable space and true class labels.
    \item \alert{Reconstruction Error} from mapping latent variables back to original image data.
  \end{itemize}
\end{frame}





\section{Methodology}
\begin{frame}{Setting: Dimension Reduction}
    Suppose we have some feature matrix $\bfY \in \bbR^{N \times D}$, containing $N$ rows of observations $\bfy_{i}^{\T} \in \bbR^{D}$.

    \bigskip

    To avoid the \emph{Curse of Dimensionality}, employ a mapping $\phi: \bbR^{D} \to \bbR^{Q}$ and work with lower-dimensional data $\phi(\bfy_{i}) \in \bbR^{Q}$.

    \bigskip

    \begin{exampleblock}{Methods for Dimension Reduction}
        \begin{enumerate}[label=\arabic*:]
            \item Principal Component Analysis (PCA)
            \item Probabilistic Principal Component Analysis (PPCA)
            \item Gaussian Process Latent Variable Models (GP-LVM)
        \end{enumerate}
    \end{exampleblock}
\end{frame}

\begin{frame}{Principal Component Analysis}
    PCA projects each observation $\bfy_{i} \in \bbR^{D}$ onto an $Q$-dimensional hyperplane which minimizes reconstruction error.

    \bigskip

    The principal components are the top $Q$ eigenvectors from the empirical covariance matrix $\hat{\bfS} \in \bbR^{D \times D}$ of $\bfY$. 
\end{frame}

\begin{frame}{Principal Component Analysis}
    \begin{exampleblock}{Benefits of PCA}
        \begin{itemize}[label=\ding{118}]
            \item Nice mathematical closed-form representation and interpretation.
            \item Relatively quick to compute.
        \end{itemize}
    \end{exampleblock}

    \bigskip

    \begin{alertblock}{Downsides of PCA}
        \begin{itemize}[label=\ding{118}]
            \item PCA enforces a \emph{linear} relationship between observations and lower-dimensional subspace.
            \item No direct probabilistic interpretation.
        \end{itemize}
    \end{alertblock}
\end{frame}


\begin{frame}{Probabilistic Principal Component Analysis}
    PPCA is similar in nature to `standard' PCA, but we make certain distributional assumptions.

    \bigskip

    \textbf{PPCA Model:} Each observation $\bfy_{i}$ is generated by latent variables $\bfu_{i}$ with an affine linear map $$\bfy_{i} = \bfW\bfu_{i} + \bmu + \bepsilon_{i}$$
\end{frame}





\section{Data Description}
\begin{frame}{\FMNIST{} Dataset}
    We use the \FMNIST{} dataset \citep{Fashion-MNIST} as our source of image data.

    \bigskip

    The dataset contains 70,000 images of fashion items (clothes and accessories) which are classified in one of ten groups. By default, \FMNIST{} is split into batches of 60,000 training images and 10,000 test images. 

    \bigskip

    Each image is a $28 \times 28$ greyscale image, represented mathematically by an observation $\bfy_{i} \in \bbR^{784}$.


\end{frame}
