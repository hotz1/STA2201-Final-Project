\section{Introduction}
\begin{frame}{Motivations \& Research Goals}
    \begin{block}{Primary Goal}
        Investigate properties of \textbf{Probabilistic Principal Component Analysis (PPCA)} and \textbf{Gaussian Process Latent Variable Models (GP-LVM)}
    \end{block}

    \bigskip

    PPCA \citep{PPCA-1999} and GP-LVM \citep{GPLVM-2005} are both extensions of principal component analysis which search for latent variable representations of high-dimensional data.
\end{frame}

\begin{frame}{Research Question}
    \begin{block}{Research Question}
        \textbf{Which latent variable methods are best suited for reducing the dimensionality of image data?}
    \end{block}
  \bigskip

  \textbf{Performance metrics:}
  \begin{itemize}[label=\ding{118}]
    \item \alert{Cluster–Label Agreement} between clusters in latent variable space and true class labels.
    \item \alert{Reconstruction Error} from mapping latent variables back to original image data.
  \end{itemize}
\end{frame}

\begin{frame}{Setting: Dimension Reduction}
    Suppose we have some feature matrix $\bfY \in \bbR^{N \times D}$, containing $N$ rows of observations $\bfy_{i}^{\T} \in \bbR^{D}$.

    \bigskip

    To avoid the \emph{Curse of Dimensionality}, employ a mapping $\phi: \bbR^{D} \to \bbR^{Q}$ and work with lower-dimensional data $\phi(\bfy_{i}) \in \bbR^{Q}$.

    \bigskip

    \begin{block}{Methods for Dimension Reduction}
        \begin{enumerate}[label=\arabic*:]
            \item Principal Component Analysis (PCA)
            \item Probabilistic Principal Component Analysis (PPCA)
            \item Gaussian Process Latent Variable Models (GP-LVM)
        \end{enumerate}
    \end{block}
\end{frame}



\section{Methods for Dimension Reduction}

\begin{frame}{Principal Component Analysis}
    PCA projects each observation $\bfy_{i} \in \bbR^{D}$ onto an $Q$-dimensional hyperplane which minimizes reconstruction error.

    \bigskip

    The principal components are the top $Q$ eigenvectors from the empirical covariance matrix ${\bfS} \in \bbR^{D \times D}$ for the feature matrix $\bfY$. 
\end{frame}

\begin{frame}{Principal Component Analysis}
    \begin{exampleblock}{Benefits of PCA}
        \begin{itemize}[label=\ding{118}]
            \item Nice mathematical closed-form representation and interpretation.
            \item Relatively quick to compute.
        \end{itemize}
    \end{exampleblock}

    \bigskip

    \begin{alertblock}{Downsides of PCA}
        \begin{itemize}[label=\ding{118}]
            \item PCA enforces a \emph{linear} relationship between observations and lower-dimensional subspace.
            \item Lacks a direct probabilistic interpretation.
        \end{itemize}
    \end{alertblock}
\end{frame}

\begin{frame}{Probabilistic Principal Component Analysis}
    PPCA is similar in nature to `standard' PCA with added distributional assumptions.

    \bigskip

    \textbf{PPCA Model:} Each observation $\bfy_{i}$ is generated by latent variables $\bfu_{i}$ according to an \emph{affine} linear mapping 
    \begin{equation*}
        \bfy_{i} = \bfW\bfu_{i} + \bmu + \bepsilon_{i}
    \end{equation*}
    \vspace*{-1.5em}
    \begin{block}{Elements of the PPCA Distribution}
        \begin{itemize}[label=\ding{118}]
            \item $\bfW \in \bbR^{D \times Q}$ 
            \item $\bmu \in \bbR^{D}$
            \item $\bfu_{i} \in \bbR^{Q}$
            \item $\bepsilon_{i} \in \bbR^{D}$, with $\bepsilon_{i} \sim \calN_{D}(\zeros, \bm{\Psi})$
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Probabilistic Principal Component Analysis}
    When $\bfu_{i} \sim \calN_{Q}(\zeros, \bfI_{Q})$ and $\operatorname{Cov}(\bepsilon_{i}) = \bm{\Psi} = \sigma^{2}\bfI_{D}$, we have that
    \begin{equation*}
        \bfy_{i} \vert \bfu_{i} \sim \calN_{D}\!\left(\bfW\bfu_{i} + \bmu, \sigma^{2}\bfI_{D}\right)
    \end{equation*}

    Integrating the conditional distribution above with respect to $\bfu_{i}$ yields the marginal distribution for the observations $\bfy_{i} \in \bbR^{Q}$
    \begin{equation*}
        \bfy_{i} \sim \calN_{D}\!\left(\bmu, \bfW\bfW^{\T} + \sigma^{2}\bfI_{D}\right)
    \end{equation*}
\end{frame}

\begin{frame}{Probabilistic Principal Component Analysis}
    The distributional form of $\bfy_{i}$ implies maximum likelihood estimators for elements of the PPCA model:
    \begin{align*}
        \bmu_{\MLE} &= \frac{1}{N}\sum_{i=1}^{N}\bfy_{i}\\
        \bfS_{\MLE} &= \frac{1}{N}\sum_{i=1}^{N}(\bfy_{i} - \bmu_{MLE})(\bfy_{i} - \bmu_{MLE})^{\T}
        \intertext{Under the PPCA modelling assumptions, we also can derive \stress{exact} closed-form equations for the MLE of $\bfW$ and $\sigma^{2}$:}
        \bfW_{\MLE} &= \bfU_{(Q)}\left(\bm{\Lambda}_{(Q)} - \sigma^{2}\right)^{\nicefrac{1}{2}}\bfR\\
        \sigma^{2}_{\MLE} &= \frac{1}{D-Q}\sum_{j=Q+1}^{D}\lambda_{j}
    \end{align*}
\end{frame}

\begin{frame}{Probabilistic Principal Component Analysis}
    \begin{exampleblock}{Benefits of PPCA}
        \begin{itemize}[label=\ding{118}]
            \item Closed-form equations for maximum likelihood estimators.
            \item Convenient probabilistic interpretation.
            \item Relatively quick to compute.
        \end{itemize}
    \end{exampleblock}

    \bigskip

    \begin{alertblock}{Downsides of PPCA}
        \begin{itemize}[label=\ding{118}]
            \item PPCA enforces a \emph{linear} relationship between the expected observations and the latent variables.
        \end{itemize}
    \end{alertblock}
\end{frame}


\begin{frame}{Gaussian Processes - Introduction}
    A \emph{Gaussian Process} $\{X_{t}\}_{t \in T}$ is a stochastic process where any finite collection of variables follows a joint multivariate Gaussian distribution.

    \bigskip 

    \begin{block}{Gaussian Process Specification}
        A Gaussian Process is fully specified by functions $\mu: T \to \bbR$ and $k: T \times T \to \bbR$ which  describe the first two moments:
        \begin{itemize}
            \item \textbf{Mean Function:} $\mu(t) = \expec{X_t}$
            \item \textbf{Covariance Function:} $k(t, t') = \covar{X_{t}}{X_{t'}}$
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Gaussian Process Regression}
    Gaussian Processes are commonly used as a model for estimating an unknown function $f: T \to \bbR$ based on observed data. 

    We place a Gaussian Process prior $f \sim \GP(\mu, k)$ on $f$ and optimize via maximum likelihood. Properties such as smoothness can be enforced via the kernel $k(\cdot, \cdot)$. 
\end{frame}

\begin{frame}{Gaussian Process Latent Variable Models}
    \textbf{Key Idea:} Using Gaussian Process regression, we can look for latent variables with non-linear relationships with the observations.

    \bigskip

    \textbf{GP-LVM Model:} Each observation $\bfy_{i}$ is generated by latent variables $\bfx_{i}$ according some unknown function $g: \bbR^{Q} \to \bbR^{D}$.
    \begin{equation*}
        \bfy_{i} = g(\bfx_{i}) + \bepsilon_{i}
    \end{equation*}    

    \begin{alertblock}{Practical Simplification}
        \begin{itemize}[label=\ding{118}]
            \item Typically, $g$ is modelled as $D$ unknown real-valued  functions; $g_{1}(\bfx), \dots, g_{D}(\bfx)$.
        \end{itemize}
    \end{alertblock}
\end{frame}

\begin{frame}{Gaussian Process Latent Variable Models}
    \begin{alertblock}{Important Notes}
        \begin{itemize}[label=\ding{118}]
            \item Instead of using a Gaussian Process regression to learn $\expec{\bfy_{i} \vert \bfx_{i}}$, GP-LVM reverses this procedure!
            \item In GP-LVM, the $Q$-dimensional latent variable representations $\{\bfx_{i}\}_{i=1}^{N}$ are learned in tandem with the functions $g_{1}, \dots, g_{D}$.
        \end{itemize}
    \end{alertblock}
\end{frame}

\begin{frame}{Gaussian Process Latent Variable Models}
    \begin{exampleblock}{Benefits of GP-LVM}
        \begin{itemize}[label=\ding{118}]
            \item Very flexible application to non-linear functions.
            \item Admits a probabilistic interpretation.
        \end{itemize}
    \end{exampleblock}

    \bigskip

    \begin{alertblock}{Downsides of GP-LVM}
        \begin{itemize}[label=\ding{118}]
            \item Highly expensive to compute, and requires iterative optimization algorithms.
            \item Does not admit a direct mapping from $\bbR^{D}$ to the latent space. 
        \end{itemize}
    \end{alertblock}
\end{frame}

\section{Data Description}
\begin{frame}{\FMNIST{} Dataset}
    We use the \FMNIST{} dataset \citep{Fashion-MNIST} as our source of image data.

    \bigskip

    The dataset contains 70,000 images of fashion items (clothes and accessories) which are classified in one of ten groups. By default, \FMNIST{} is split into batches of 60,000 training images and 10,000 test images. 

    \bigskip

    Each image is a $28 \times 28$ greyscale image, represented mathematically by an observation $\bfy_{i} \in \bbR^{784}$.


\end{frame}

\begin{frame}{Selection of $Q$}
    The Scree plots do not depict strong `elbows' where we can clearly delineate important variables. 

    \bigskip

    \begin{block}{Cumulative Variance Explained}
        \begin{itemize}[label=\ding{118}]
            \item The first 49 components explain 80\% of the variance in the data.
            \item The first 136 components explain 90\% of the variance in the data.
        \end{itemize}
    \end{block}

    \bigskip

    \begin{alertblock}{Final Selection of $Q$}
        \begin{itemize}[label=\ding{118}]
            \item For visualizations and interpretations, we use $Q = 2$ and $Q = 3$. 
            \item For creating our models, we use $Q = 10$. 
        \end{itemize}
    \end{alertblock}    
\end{frame}

\begin{frame}{Model Comparison}
    To reduce the overall computational load, we reduce the dataset $\bfY^{\train}$ from 60,000 observations to 10,000 observations. 
    
    We stratify the sub-sampled data in order to keep the proportions of labels the same in the reduced dataset.

    For all three model types (PCA, PPCA, GP-LVM), we reduce the sub-sampled data to $Q = 10$ dimensions, and use $K$-means to cluster data in the reduced space.
\end{frame}

\section{Discussion}
\begin{frame}{Limitations}
    \begin{alertblock}{Computational Costs}
        \begin{itemize}[label=\ding{118}]
            \item Inferring the latent matrix $\bfX \in \bbR^{N \times Q}$ is \stress{incredibly} expensive, limiting scalability. 
            \item We had to reduce the size of both $N$ and $Q$ to perform the analysis, and we were unable to perform out-of-sample prediction for the GP-LVM model.
        \end{itemize}
    \end{alertblock}
\end{frame}

\begin{frame}{Further Research}
    There are several avenues for future research to both overcome the mentioned limitations and to further scale this project. 

    \bigskip

    \begin{exampleblock}{Future Work}
        \begin{itemize}[label=\ding{118}]
            \item Working with different values of $Q$ (particularly larger $Q$) to assess differences in performance.
            \item Incorporate out-of-sample predictions using $\bfY^{\test}$.
            \item Utilize more refined GP-LVM methods for improved performance.
        \end{itemize}
    \end{exampleblock}
\end{frame}
